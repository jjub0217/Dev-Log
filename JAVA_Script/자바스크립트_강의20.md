```js
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
</head>

<body>
    <input class="input_todo" type="text" placeholder="enter todo!">
    <ul class="todos"></ul>
    <script>
        const $input = document.querySelector('.input_todo');
        // document.querySelector('.input_todo').onkeyup = function () { } // <- 이벤트 핸들러 등록
        const $todos = document.querySelector('.todos'); // <- HTML 마크업 순서대로 해야 하니까 여기다가 $ul.innerHTML 해야 한다.

        $input.onkeyup = function (e) {
            // <- 여기다가 $ul.innerHTML 하면 안됨.
            if (e.keyCode !== 13) return;

            console.log($input.value); // <- li 만드는데 필요한것이었다.
            $input.value.trim();
            $todos.innerHTML += `<li>${$input.value}</li>`;
            $input.value = ''; // <- 지우는거다.
        };
    </script>
</body>

</html>
```

로드 이벤트에 이벤트 핸들러를 달면 무슨 뜻이지?

```js
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
</head>

<body>
    <input class="input_todo" type="text" placeholder="enter todo!">
    <ul class="todos"></ul>
    <script>
        let todos = [];

        const $input = document.querySelector('.input_todo');
        // document.querySelector('.input_todo').onkeyup = function () { } // <- 이벤트 핸들러 등록
        const $todos = document.querySelector('.todos'); // <- HTML 마크업 순서대로 해야 하니까 여기다가 $ul.innerHTML 해야 한다.

        window.onload = () => {
            todos = [
            { id: 1, content: 'HTML', completed: false },
            { id: 2, content: 'CSS', completed: true },
            { id: 3, content: 'Javascript', completed: false }
            ];
        };

        $input.onkeyup = function (e) {
            // <- 여기다가 $ul.innerHTML 하면 안됨.
            if (e.keyCode !== 13) return;
            // console.log($input.value); // <- li 만드는데 필요한것이었다.
            $input.value = $input.value.trim();
            if ($input.value) $todos.innerHTML += `<li>${$input.value}</li>`;
            // $todos.innerHTML += `<li>${$input.value}</li>`;
            $input.value = ''; // <- 지우는거다.
        };
    </script>
</body>

</html>
```
